QString IcCardTool::unsignedToQString(unsigned char *tmp, int len)
{
    QString msg = "";
    for (int i = 0; i < len; ++i)
    {
        msg += QString("%1").arg(int(tmp[i]), 2, 16, QChar('0')).toUpper();
        if (i != len - 1)
        {
            msg += ' ';
        }
    }
    return msg;
}
